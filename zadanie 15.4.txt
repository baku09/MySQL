SELECT I.*, IL.NAME AS LIST_NAME
FROM ISSUES I
JOIN ISSUESLISTS IL on IL.ID = I.ISSUESLIST_ID;

Zad.2
SELECT I.*, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID;

Zad.3
SELECT U.FIRSTNAME, U.LASTNAME, COUNT(I.ID) AS NUMBER_TASKS
FROM USERS U
JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID, U.FIRSTNAME, U.LASTNAME;
